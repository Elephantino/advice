строки

cтрока - примитив, приметив неизменяем(только заменяем), по этому все методы возвращают новую строку а не изменяют старую

доступ к символам подобен массивам

перебор строк подобен массивам

если на месте искомой позиции символа нет

charAt - вернет пустую строку ''

[ ] – вернет undefined

String.prototype.methods


let str = "Qwe";
let str2 = "rty";

                         
str.at(0)                | --- TypeError | символ по индексу
str.charAt(0)            | Q             | символ по индексу
str.indexOf("we")        | 1             | индекс вхождения
str.lastIndexOf("w")     | 1             | индекс вхождения

str.concat(str2)         | Qwerty        | строка из множества строк
str.repeat(2)            | QweQwe        | строка из повторов
str.slice(0, 2)          | Qw            | кусок строки по индексам
str.substr(1, 2)         | we            | кусок строки по индексам

 
str.match()              |               | возвращает совпадения с регулярным выражением
str.replace()            |               | вернет строку, заменит часть
str.search()             |               | индекс первого совпадения с регулярным выражением


str.includes("Qwe")      | true          | содержит ли строка заданную подстроку
str.startWith("Qwe")     | --- TypeError | начинается ли строка с символов
